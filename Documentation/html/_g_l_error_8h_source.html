11 * This class provides functionality to check for OpenGL errors after

13 * logging system. The class can be used
14 * in a RAII manner to automatically check for errors when going out of scope.
15 *
16 * Usage:
17 * @code
18 * {
19 * GLErrorChecker glError("Rendering Frame");
20 * // OpenGL calls...
21 * } // Errors checked automatically here
22 * @endcode
23 * @note This class is intended for debugging purposes and may be disabled
24 * in release builds for performance.
25 */

39 /// @brief Checks for OpenGL errors and logs them.
40 void CheckErrors() const {
41 GLenum error;
42 bool hasError = false;
43 std::string errorMessages;
44
45 while ((error = glGetError()) != GL_NO_ERROR) {
46 hasError = true;
47 const char* errorString = GetErrorString(error);
48 if (!errorMessages.empty()) errorMessages += ", ";
49 errorMessages += errorString;
50 }
51
52 if (hasError) {
53 ENGINE_ERROR << "OpenGL Error in " << m_Context << ": " << errorMessages;
54 }
55 }

57 /// @brief Converts an OpenGL error code to a human-readable string.
58 static const char* GetErrorString(GLenum error) {
59 switch (error) {
60 case GL_NO_ERROR: return "GL_NO_ERROR";
61 case GL_INVALID_ENUM: return "GL_INVALID_ENUM";
62 case GL_INVALID_VALUE: return "GL_INVALID_VALUE";
63 case GL_INVALID_OPERATION: return "GL_INVALID_OPERATION";
64 case GL_OUT_OF_MEMORY: return "GL_OUT_OF_MEMORY";
65 case GL_INVALID_FRAMEBUFFER_OPERATION: return "GL_INVALID_FRAMEBUFFER_OPERATION";
66 default: return "UNKNOWN_ERROR";
67 }
68 }
69
70private:
