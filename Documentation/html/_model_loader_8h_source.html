10/**
11 * @class ModelLoader
12 * @brief Utility class for loading 3D models and meshes from files.
13 *
14 * This class uses the Assimp library to load 3D model files in various formats
15 * (e.g., OBJ, FBX, STL) and converts them into Mesh objects usable by the engine.
16 * It handles parsing the model file, extracting vertex and index data, and
17 * creating Mesh instances with associated materials.
18 *
19 * Key features:
20 * - Load single meshes or entire models with multiple meshes
21 * - Support for common 3D file formats via Assimp
22 * - Automatic processing of vertices, indices, normals, and texture coordinates
23 *
24 * @note Requires Assimp library to be linked with the project.
25 * @note Currently does not handle animations or skeletal data.
26 *
27 * @example
28 * @code
29 * Ref<Mesh> mesh = ModelLoader::LoadMesh("path/to/model.obj");
30 * std::vector<Ref<Mesh>> modelMeshes = ModelLoader::LoadModel("path/to/model.fbx");
31 * @endcode
32 */
33class ModelLoader {
34public:
35 /**
36 * @brief Loads a single mesh from the specified file.
37 *
38 * Uses Assimp to import the model file and extract the first mesh found.
39 * Converts the mesh data into a Mesh object.
40 *
41 * @param filepath The path to the model file to load
42 * @return Ref<Mesh> The loaded mesh object
43 *
44 * @note If the file contains multiple meshes, only the first one is loaded.
45 * @note Throws an exception if loading fails.
46 */
47 static Ref<Mesh> LoadMesh(const std::string& filepath);

54 *
55 * @param filepath The path to the model file to load
56 * @return std::vector<Ref<Mesh>> A vector of loaded mesh objects

58 * @note Supports models with multiple meshes.

61 static std::vector<Ref<Mesh>> LoadModel(const std::string& filepath);
