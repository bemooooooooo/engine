35void Scene::RemoveObject(const std::string& name) {
36 auto indexIt = m_ObjectIndex.find(name);
37 if (indexIt == m_ObjectIndex.end()) {
38 ENGINE_WARN << "Object not found in scene: " << name;
39 return;
40 }
41
42 size_t index = indexIt->second;
43
44 // Удаляем из вектора
45 m_Objects.erase(m_Objects.begin() + index);
46
47 // Обновляем индекс для всех объектов после удаленного
48 m_ObjectIndex.erase(indexIt);
49 for (auto& pair : m_ObjectIndex) {
50 if (pair.second > index) {
51 pair.second--;
52 }
53 }
54
55 ENGINE_INFO << "Removed object from scene: " << name;
56}
