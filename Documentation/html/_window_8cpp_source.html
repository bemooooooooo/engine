24bool Window::Initialize(const WindowProps& props) {
25 ENGINE_INFO << "Creating window: " << props.Title << " (" << props.Width << "x" << props.Height << ")";
26
27 m_Data.Title = props.Title;
28 m_Data.Width = props.Width;
29 m_Data.Height = props.Height;
30
31 // Инициализация GLFW
32 if (!s_GLFWInitialized) {
33 int success = glfwInit();
34 if (!success) {
35 ENGINE_CRITICAL << "Could not initialize GLFW!";
36 return false;
37 }
38
39 glfwSetErrorCallback(GLFWErrorCallback);
40 s_GLFWInitialized = true;
41 ENGINE_INFO << "GLFW initialized";
42 }
43
44 // Настройка OpenGL context
45 glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 3);
46 glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 3);
47 glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);
48 #ifdef __APPLE__
49 glfwWindowHint(GLFW_OPENGL_FORWARD_COMPAT, GL_TRUE);
50 #endif
51
52 // Создание окна
53 m_Window = glfwCreateWindow(
54 static_cast<int>(m_Data.Width),
55 static_cast<int>(m_Data.Height),
56 m_Data.Title.c_str(),
57 nullptr, nullptr
58 );
59
60 if (!m_Window) {
61 ENGINE_CRITICAL << "Failed to create GLFW window!";
62 glfwTerminate();
63 return false;
64 }
65
66 // Сделать контекст текущим
67 glfwMakeContextCurrent(static_cast<GLFWwindow*>(m_Window));
68
69 // Инициализация Glad
70 if (!gladLoadGL(glfwGetProcAddress)) {
71 ENGINE_CRITICAL << "Failed to initialize Glad!";
72 return false;
73 }
74
75 ENGINE_INFO << "OpenGL Info:";
76 ENGINE_INFO << " Vendor: " << glGetString(GL_VENDOR);
77 ENGINE_INFO << " Renderer: " << glGetString(GL_RENDERER);
78 ENGINE_INFO << " Version: " << glGetString(GL_VERSION);
79
80 // Настройка колбэков
81 SetupCallbacks();
82
83 // Базовая настройка OpenGL
84 glEnable(GL_DEPTH_TEST);
85 glClearColor(0.1f, 0.1f, 0.1f, 1.0f);
86
87 ENGINE_INFO << "Window created successfully";
88 return true;
89}

114void Window::SetupCallbacks() {
115 GLFWwindow* window = static_cast<GLFWwindow*>(m_Window);
116
117 // Сохраняем указатель на данные окна для доступа в статических колбэках
118 glfwSetWindowUserPointer(window, &m_Data);
119
120 // Колбэк изменения размера окна
121 glfwSetWindowSizeCallback(window, [](GLFWwindow* window, int width, int height) {
122 WindowData& data = *static_cast<WindowData*>(glfwGetWindowUserPointer(window));
123 data.Width = width;
124 data.Height = height;
125
126 // Создаем и передаем событие через EventCallback
127 WindowResizeEvent event(static_cast<u32>(width), static_cast<u32>(height));
128 if (data.EventCallback) {
129 data.EventCallback(event);
130 }
131 });
132
133 // Колбэк закрытия окна
134 glfwSetWindowCloseCallback(window, [](GLFWwindow* window) {
135 WindowData& data = *static_cast<WindowData*>(glfwGetWindowUserPointer(window));
136
137 // Создаем и передаем событие через EventCallback
138 WindowCloseEvent event;
139 if (data.EventCallback) {
140 data.EventCallback(event);
141 }
142 });
143}
