7void GraphicsContext::Init() {
8 if (s_Initialized) return;
9
10 // Проверяем что Glad инициализирован
11 if (!gladLoadGL((GLADloadfunc)glfwGetProcAddress)) {
12 ENGINE_CRITICAL << "Glad not initialized!";
13 return;
14 }
15
16 // Базовая настройка OpenGL
17 EnableDepthTest(true);
18 EnableBlending(true);
19 EnableCulling(true);
20 SetCullFace(true);
21
22 // Настройка blending
23 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
24
25 // Настройка глубины
26 glDepthFunc(GL_LEQUAL);
27 glClearDepth(1.0f);
28
29 // Проверка состояния после инициализации
30 GLint depthTest, depthFunc, cullFace, blendEnabled;
31 glGetIntegerv(GL_DEPTH_TEST, &depthTest);
32 glGetIntegerv(GL_DEPTH_FUNC, &depthFunc);
33 glGetIntegerv(GL_CULL_FACE, &cullFace);
34 glGetIntegerv(GL_BLEND, &blendEnabled);
35
36 ENGINE_INFO << "OpenGL State after init:";
37 ENGINE_INFO << " Depth Test: " << (depthTest ? "ON" : "OFF");
38 ENGINE_INFO << " Depth Func: 0x" << std::hex << depthFunc;
39 ENGINE_INFO << " Face Culling: " << (cullFace ? "ON" : "OFF");
40 ENGINE_INFO << " Blending: " << (blendEnabled ? "ON" : "OFF");
41
42 s_Initialized = true;
43
44 ENGINE_INFO << "Graphics Context Initialized";
45 ENGINE_INFO << " Vendor: " << GetVendor();
46 ENGINE_INFO << " Renderer: " << GetRenderer();
47 ENGINE_INFO << " Version: " << GetVersion();
48}
