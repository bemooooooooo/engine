9/**
10 * @struct SceneObject
11 * @brief Represents an object within a 3D scene, containing a mesh and transform.
12 *
13 * SceneObject encapsulates a 3D mesh along with its transformation (position,
14 * rotation, scale) in the scene. It also supports an optional update callback
15 * for per-frame logic.
16 *
17 * Key features:
18 * - Holds a reference to a Mesh for rendering
19 * - Contains a Transform for spatial positioning
20 * - Supports an onUpdate callback for custom behavior each frame
21 *
22 * @note The onUpdate function, if set, is called during the scene's Update()
23 * method with the delta time parameter.
24 *
25 * @example
26 * @code
27 * Ref<Mesh> cubeMesh = ResourceManager::GetMesh("cube");
28 * SceneObject cubeObject("Cube", cubeMesh);
29 * cubeObject.transform.position = Vector3(0, 0, 0);
30 * cubeObject.onUpdate = [](SceneObject* obj, f32 deltaTime) {
31 * obj->transform.rotation.y += 90.0f * deltaTime; // Rotate 90 degrees per second
32 * };
33 * @endcode
34 */
35struct SceneObject {
36 /**
37 * @var name
38 * @brief The unique name identifier for the scene object.
39 */
40 std::string name;
41
42 /**
43 * @var mesh
44 * @brief The mesh associated with this scene object.
45 */
46 Ref<Mesh> mesh;
47
48 /**
49 * @var transform
50 * @brief The transformation (position, rotation, scale) of the object in the scene.
51 */
52 Transform transform;
53
54 /**
55 * @var onUpdate
56 * @brief Optional callback function called each frame during scene update.
57 *
58 * This function, if set, is invoked during the Scene's Update() method,
59 * allowing for custom per-frame logic such as animations or behavior.
60 * The function receives a pointer to the SceneObject and the delta time
61 * since the last frame.
62 *
63 * @note Set to nullptr by default (no update).
64 *
65 * @example
66 * @code
67 * sceneObject.onUpdate = [](SceneObject* obj, f32 deltaTime) {
68 * obj->transform.position.x += 1.0f * deltaTime; // Move right at 1 unit/sec
69 * };
70 * @endcode
71 */
72 std::function<void (SceneObject*, f32)> onUpdate = nullptr;
