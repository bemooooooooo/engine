7/**
8 * @struct MaterialProperties
9 * @brief Encapsulates the properties and texture maps of a material.
10 *
11 * This structure holds the various properties that define the appearance
12 * of a material in the rendering engine, including color, metallicity,
13 * roughness, ambient occlusion, and associated texture maps.
14 *
15 * Properties:
16 * - albedo: Base color of the material
17 * - metallic: Metallic factor (0.0 = non-metal, 1.0 = metal)
18 * - roughness: Surface roughness (0.0 = smooth, 1.0 = rough)
19 * - ambientOcclusion: Ambient occlusion factor
20 *
21 * Texture Maps:
22 * - albedoMap: Texture for base color
23 * - normalMap: Texture for surface normals
24 * - metallicMap: Texture for metallic values
25 * - roughnessMap: Texture for roughness values
26 * - ambientOcclusionMap: Texture for ambient occlusion values
27 *
28 * @example
29 * @code
30 * MaterialProperties props;
31 * props.albedo = Vector4(1.0f, 0.0f, 0.0f, 1.0f); // Red color
32 * props.metallic = 0.5f;
33 * props.roughness = 0.2f;
34 * props.albedoMap = CreateRef<Texture>("textures/wood_albedo.png");
35 * @endcode
36 */
37struct MaterialProperties {
38 // Default material properties
39 Vector4 albedo = Vector4(1.0f, 1.0f, 1.0f, 1.0f);
40 f32 metallic = 0.0f;
41 f32 roughness = 0.5f;
42 f32 ambientOcclusion = 1.0f;
43
44 // Texture maps
45 Ref<Texture> albedoMap;
46 Ref<Texture> normalMap;
47 Ref<Texture> metallicMap;
48 Ref<Texture> roughnessMap;
49 Ref<Texture> ambientOcclusionMap;
