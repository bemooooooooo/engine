74/**
75 * @brief Clamps a value between a minimum and maximum.
76 *
77 * Returns the value limited to the specified range [min, max].
78 * If value < min, returns min. If value > max, returns max.
79 * Otherwise, returns value.
80 *
81 * @tparam T The type of the value (must support comparison operators)
82 * @param value The value to clamp
83 * @param min The minimum allowable value
84 * @param max The maximum allowable value
85 * @return T The clamped value within [min, max]
86 *
87 * @example
88 * @code
89 * f32 clamped = Math::Clamp(5.0f, 0.0f, 3.0f); // clamped = 3.0f
90 * f32 clamped2 = Math::Clamp(-1.0f, 0.0f, 3.0f); // clamped2 = 0.0f
91 * f32 clamped3 = Math::Clamp(2.0f, 0.0f, 3.0f); // clamped3 = 2.0f
92 * @endcode
93 */
94template <typename T>
95constexpr T Clamp(T value, T min, T max) {
96 return (value < min) ? min : (value > max) ? max : value;
97}

99/**
100 * @brief Linearly interpolates between two values.
101 *
102 * Performs linear interpolation (lerp) between a and b using parameter t.
103 * Formula: result = a + (b - a) * t
104 *
105 * @tparam T The type of the values (must support arithmetic operators)
106 * @param a The start value (t = 0)
107 * @param b The end value (t = 1)
108 * @param t The interpolation parameter (0.0 to 1.0)
109 * @return T The interpolated value
110 *
111 * Parameter t behavior:
112 * - t = 0.0: Returns a
113 * - t = 1.0: Returns b
114 * - t < 0.0 or t > 1.0: Extrapolates beyond a and b
115 *
116 * @example
117 * @code
118 * f32 result = Math::Lerp(0.0f, 10.0f, 0.5f); // result = 5.0f
119 * f32 result2 = Math::Lerp(20.0f, 30.0f, 0.25f); // result2 = 22.5f
120 * @endcode
121 */
122template <typename T>
123constexpr T Lerp(T a, T b, f32 t) {
124 return a + t * (b - a);
125}

337/**
338 * @brief Smoothstep function for smooth interpolation.
339 *
340 * Performs smooth Hermite interpolation between 0 and 1 when x is in [edge0, edge1].
341 * The function returns 0 if x <= edge0 and 1 if x >= edge1.
342 * The interpolation has zero first derivatives at the endpoints.
343 *
344 * @param edge0 The lower edge of the transition
345 * @param edge1 The upper edge of the transition
346 * @param x The input value to interpolate
347 * @return f32 The smoothly interpolated value between 0 and 1
348 * @example
349 * @code
350 * f32 smooth = Math::Smoothstep(0.0f, 1.0f, 0.5f); // smooth = 0.5f
351 * @endcode
352 */
353inline f32 Smoothstep(f32 edge0, f32 edge1, f32 x) {
354 x = Clamp((x - edge0) / (edge1 - edge0), 0.0f, 1.0f);
355 return x * x * (3.0f - 2.0f * x);
356}

358/**
359 * @brief Smootherstep function for even smoother interpolation.
360 *
361 * Performs smoother Hermite interpolation between 0 and 1 when x is in [edge0, edge1].
362 * The function returns 0 if x <= edge0 and 1 if x >= edge1.
363 * The interpolation has zero first and second derivatives at the endpoints.
364 *
365 * @param edge0 The lower edge of the transition
366 * @param edge1 The upper edge of the transition
367 * @param x The input value to interpolate
368 * @return f32 The smoothly interpolated value between 0 and 1
369 *
370 * @example
371 * @code
372 * f32 smoother = Math::Smootherstep(0.0f, 1.0f, 0.5f); // smoother = 0.5f
373 * @endcode
374 */
375inline f32 Smootherstep(f32 edge0, f32 edge1, f32 x) {
376 x = Clamp((x - edge0) / (edge1 - edge0), 0.0f, 1.0f);
377 return x * x * x * (x * (x * 6.0f - 15.0f) + 10.0f);
378}

384 *
385 * @param value The input value to remap
386 * @param fromMin The lower bound of the input range
387 * @param fromMax The upper bound of the input range
388 * @param toMin The lower bound of the output range
389 * @param toMax The upper bound of the output range

397inline f32 Remap(f32 value, f32 fromMin, f32 fromMax, f32 toMin, f32 toMax) {

458/**

469constexpr T NextPowerOfTwo(T value) {

471 value--;
472 value |= value >> 1;
473 value |= value >> 2;
474 value |= value >> 4;
475 value |= value >> 8;
476 value |= value >> 16;
