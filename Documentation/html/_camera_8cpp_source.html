102void Camera::Recalculate() {
103 // Recalculate view matrix
104 Vector3 forward = GetForward();
105 Vector3 up = GetUp();
106
107 m_ViewMatrix = Matrix4::LookAt(m_Position, m_Position + forward, up);
108
109 // Recalculate projection matrix
110 if (m_Type == CameraType::Perspective) {
111 m_ProjectionMatrix = Matrix4::Perspective(m_FOV, m_Aspect, m_Near, m_Far);
112 } else {
113 f32 right = m_Size * m_Aspect * 0.5f;
114 f32 left = -right;
115 f32 top = m_Size * 0.5f;
116 f32 bottom = -top;
117 m_ProjectionMatrix = Matrix4::Orthographic(left, right, bottom, top, m_Near, m_Far);
118 }
