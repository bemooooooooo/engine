20Ref<Shader> ResourceManager::LoadShader(const std::string& name,
21 const std::string& vertexPath,
22 const std::string& fragmentPath) {
23 // Проверяем, не загружен ли уже
24 if (s_Shaders.find(name) != s_Shaders.end()) {
25 ENGINE_WARN << "Shader '" << name << "' already loaded, returning existing";
26 return s_Shaders[name];
27 }
28
29 auto shader = Shader::CreateFromFiles(vertexPath, fragmentPath);
30 if (shader) {
31 s_Shaders[name] = shader;
32 ENGINE_INFO << "Shader '" << name << "' loaded successfully";
33 } else {
34 ENGINE_ERROR << "Failed to load shader '" << name << "'";
35 }
36
37 return shader;
