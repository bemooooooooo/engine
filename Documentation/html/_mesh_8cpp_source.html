132Ref<Mesh> Mesh::CreateSphere(u32 segments) {
133 std::vector<Vertex> vertices;
134 std::vector<u32> indices;
135
136 segments = Math::Max(segments, 3u);
137
138 // Generate vertices
139 for (u32 y = 0; y <= segments; y++) {
140 for (u32 x = 0; x <= segments; x++) {
141 f32 xSegment = (f32)x / (f32)segments;
142 f32 ySegment = (f32)y / (f32)segments;
143
144 f32 xPos = std::cos(xSegment * 2.0f * Math::PI) * std::sin(ySegment * Math::PI);
145 f32 yPos = std::cos(ySegment * Math::PI);
146 f32 zPos = std::sin(xSegment * 2.0f * Math::PI) * std::sin(ySegment * Math::PI);
147
148 Vertex vertex;
149 vertex.position = Vector3(xPos, yPos, zPos) * 0.5f;
150 vertex.normal = Vector3(xPos, yPos, zPos).Normalized();
151 vertex.texCoord = Vector2(xSegment, ySegment);
152
153 vertices.push_back(vertex);
154 }
155 }
156
157 // Generate indices
158 for (u32 y = 0; y < segments; y++) {
159 for (u32 x = 0; x < segments; x++) {
160 u32 first = (y * (segments + 1)) + x;
161 u32 second = first + segments + 1;
162
163 indices.push_back(first);
164 indices.push_back(second);
165 indices.push_back(first + 1);
166
167 indices.push_back(second);
168 indices.push_back(second + 1);
169 indices.push_back(first + 1);
170 }
171 }
172
173 // Calculate tangents and bitangents
174 for (size_t i = 0; i < indices.size(); i += 3) {
175 u32 i0 = indices[i];
176 u32 i1 = indices[i + 1];
177 u32 i2 = indices[i + 2];
178
179 Vertex& v0 = vertices[i0];
180 Vertex& v1 = vertices[i1];
181 Vertex& v2 = vertices[i2];
182
183 Vector3 edge1 = v1.position - v0.position;
184 Vector3 edge2 = v2.position - v0.position;
185
186 Vector2 deltaUV1 = v1.texCoord - v0.texCoord;
187 Vector2 deltaUV2 = v2.texCoord - v0.texCoord;
188
189 f32 f = 1.0f / (deltaUV1.x * deltaUV2.y - deltaUV2.x * deltaUV1.y);
190
191 Vector3 tangent;
192 tangent.x = f * (deltaUV2.y * edge1.x - deltaUV1.y * edge2.x);
193 tangent.y = f * (deltaUV2.y * edge1.y - deltaUV1.y * edge2.y);
194 tangent.z = f * (deltaUV2.y * edge1.z - deltaUV1.y * edge2.z);
195 tangent = tangent.Normalized();
196
197 Vector3 bitangent;
198 bitangent.x = f * (-deltaUV2.x * edge1.x + deltaUV1.x * edge2.x);
199 bitangent.y = f * (-deltaUV2.x * edge1.y + deltaUV1.x * edge2.y);
200 bitangent.z = f * (-deltaUV2.x * edge1.z + deltaUV1.x * edge2.z);
201 bitangent = bitangent.Normalized();
202
203 v0.tangent = tangent;
204 v1.tangent = tangent;
205 v2.tangent = tangent;
206
207 v0.bitangent = bitangent;
208 v1.bitangent = bitangent;
209 v2.bitangent = bitangent;
210 }
211
212 return CreateRef<Mesh>(vertices, indices);
213}
