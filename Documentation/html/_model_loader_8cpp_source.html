31void ModelLoader::ProcessNode(aiNode* node, const aiScene* scene,
32 std::vector<Ref<Mesh>>& meshes) {
33 // Process each mesh located at the current node
34 for (u32 i = 0; i < node->mNumMeshes; i++) {
35 aiMesh* mesh = scene->mMeshes[node->mMeshes[i]];
36 meshes.push_back(ProcessMesh(mesh));
37 }
38
39 // Process each of the children nodes
40 for (u32 i = 0; i < node->mNumChildren; i++) {
41 ProcessNode(node->mChildren[i], scene, meshes);
42 }
43}

51std::vector<Vertex> ModelLoader::LoadVertices(aiMesh* mesh) {
52 std::vector<Vertex> vertices;
53 vertices.reserve(mesh->mNumVertices);
54
55 for (u32 i = 0; i < mesh->mNumVertices; i++) {
56 Vertex vertex;
57
58 // Position
59 vertex.position.x = mesh->mVertices[i].x;
60 vertex.position.y = mesh->mVertices[i].y;
61 vertex.position.z = mesh->mVertices[i].z;
62
63 // Normal
64 if (mesh->HasNormals()) {
65 vertex.normal.x = mesh->mNormals[i].x;
66 vertex.normal.y = mesh->mNormals[i].y;
67 vertex.normal.z = mesh->mNormals[i].z;
68 }
69
70 // Texture coordinates
71 if (mesh->mTextureCoords[0]) {
72 vertex.texCoord.x = mesh->mTextureCoords[0][i].x;
73 vertex.texCoord.y = mesh->mTextureCoords[0][i].y;
74 }
75
76 // Tangents and bitangents
77 if (mesh->HasTangentsAndBitangents()) {
78 vertex.tangent.x = mesh->mTangents[i].x;
79 vertex.tangent.y = mesh->mTangents[i].y;
80 vertex.tangent.z = mesh->mTangents[i].z;
81
82 vertex.bitangent.x = mesh->mBitangents[i].x;
83 vertex.bitangent.y = mesh->mBitangents[i].y;
84 vertex.bitangent.z = mesh->mBitangents[i].z;
85 }
86
87 vertices.push_back(vertex);
88 }
89
90 return vertices;
91}
