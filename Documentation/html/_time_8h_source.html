14namespace Time {

16/**
17 * @class Timer
18 * @brief High-precision timer for measuring elapsed time and frame delta times.
19 *
20 * This class provides a flexible timer that can measure both total elapsed time
21 * since creation/reset and frame-to-frame delta times. It uses the system's
22 * high-resolution clock for accurate timing measurements needed in game loops.
23 *
24 * The timer is useful for:
25 *
26 * - Frame timing and delta time calculation
27 *
28 * - Performance profiling and benchmarking
29 *
30 * - Game logic that depends on elapsed time
31 *
32 * - Animations and transitions
33 *
34 * - Input response timing
35 *
36 * @note This class is not thread-safe. Use separate Timer instances for different threads.
37 *
38 * @example
39 * @code
40 * Time::Timer gameTimer;
41 *
42 * while (gameRunning) {
43 * f32 deltaTime = gameTimer.Tick(); // Get time since last frame
44 * gameLogic.Update(deltaTime);
45 *
46 * if (gameTimer.Elapsed() > 60.0f) { // After 60 seconds
47 * GameOver();
48 * }
49 * }
50 * @endcode
51 */
52class Timer {
53 public:
54 /**
55 * @typedef Clock
56 * @brief High-resolution clock type used for timing measurements.
57 *
58 * Uses std::chrono::high_resolution_clock which provides the highest
59 * available resolution for time measurements on the system.
60 */
61 using Clock = std::chrono::high_resolution_clock;
62
63 /**
64 * @typedef TimePoint
65 * @brief Type representing a specific point in time from the high-resolution clock.
66 *
67 * Represents a single moment in time with the precision of the high-resolution clock.
68 * Used internally to track timer start time and last frame time.
69 */
70 using TimePoint = std::chrono::time_point<Clock>;
71
72 /**
73 * @typedef Duration
74 * @brief Type representing a duration of time in floating-point seconds.
75 *
76 * A duration type that measures time intervals in floating-point seconds,
77 * providing fractional second precision for frame delta times and other timing needs.
78 */
79 using Duration = std::chrono::duration<f32>;
80
81 /**
82 * @brief Constructs a new Timer instance and starts it immediately.
83 *
84 * The constructor initializes the timer with the current time as the start time.
85 * The timer begins measuring immediately upon construction.
86 *
87 * @note The timer tracks both total elapsed time and frame-to-frame delta times.
88 *
89 * @example
90 * @code
91 * Time::Timer timer; // Timer starts now
92 * @endcode
93 */
94 Timer();
95
96 /**
97 * @brief Resets the timer to its initial state.
98 *
99 * When called, this function sets the start time of the timer to the current time,
100 * effectively restarting the timer. The elapsed time becomes 0, and subsequent
101 * calls to Elapsed() will measure time from this reset point.
102 *
103 * This is useful when you want to:
104 *
105 * - Restart timing for a new section
106 *
107 * - Reset timing between game phases
108 *
109 * - Clear accumulated timing for fresh measurements
110 *
111 * @note This does not affect the last frame time used by Tick().
112 *
113 * @example
114 * @code
115 * timer.Reset();
116 * assert(timer.Elapsed() < 0.001f); // Should be very close to 0
117 * @endcode
118 */
119 void Reset();
120
121 /**
122 * @brief Returns the total time elapsed since timer start or last reset.
123 *
124 * This function returns the cumulative time that has passed since the timer was
125 * constructed or Reset() was called. The returned value is in seconds as a
126 * floating-point number, allowing for fractional second precision.
127 *
128 * @return f32 The elapsed time in seconds (e.g., 1.5 = 1 second and 500 milliseconds)
129 *
130 * @note This is a const function and does not modify the timer state.
131 * @note Call Reset() to reset this counter.
132 *
133 * @example
134 * @code
135 * Time::Timer timer;
136 * // ... do some work ...
137 * f32 elapsedSeconds = timer.Elapsed();
138 * ENGINE_DEBUG_LOG("Operation took: " << elapsedSeconds << " seconds");
139 * @endcode
140 */
141 f32 Elapsed() const;
142
143 /**
144 * @brief Calculates and returns the time elapsed since the last Tick() call.
145 *
146 * This function calculates the delta time (time difference) since the last time
147 * Tick() was called. On the first call, it returns the time since the timer was
148 * constructed or Reset(). Subsequent calls return the frame-to-frame delta time.
149 *
150 * This is the primary function used for frame-rate independent game logic updates.
151 * It allows game physics, movement, and animations to advance proportionally to
152 * the time that has actually passed, not the number of frames rendered.
153 *
154 * @return f32 The time since last Tick() call in seconds
155 *
156 * @note This function updates the internal last-frame timestamp.
157 * @note The returned delta time includes all system overhead (rendering, input, etc.)
158 *
159 * @example
160 * @code
161 * Time::Timer timer;
162 *
163 * while (gameRunning) {
164 * f32 deltaTime = timer.Tick(); // Get time since last frame
165 *
166 * // Update game logic with frame-rate independent timing
167 * player.position += player.velocity * deltaTime;
168 * particle.lifetime -= deltaTime;
169 *
170 * // Render frame...
171 * }
172 * @endcode
173 *
174 * @example Usage in physics:
175 * @code
176 * f32 deltaTime = gameTimer.Tick();
177 * entity.velocity.y -= gravity * deltaTime; // Frame-rate independent gravity
178 * entity.position += entity.velocity * deltaTime;
179 * @endcode
180 */
181 f32 Tick();
182
183 private:
184 /**
185 * @var m_StartTime_
186 * @brief The time point when the timer was created or last reset.
187 *
188 * Used to calculate total elapsed time via the Elapsed() function.
189 * Updated when Reset() is called.
190 */
191 TimePoint m_StartTime_;
192
193 /**
194 * @var m_LastFrameTime_
195 * @brief The time point of the last Tick() call.
196 *
197 * Used to calculate the delta time (time between frames) in the Tick() function.
198 * Updated after each Tick() call.
199 */
200 TimePoint m_LastFrameTime_;
201};
