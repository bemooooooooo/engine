5 // Create default PBR shader
6 const char* defaultVertexShader = R"(
7 #version 330 core
8 layout (location = 0) in vec3 a_Position;
9 layout (location = 1) in vec3 a_Normal;
10 layout (location = 2) in vec2 a_TexCoord;
11 layout (location = 3) in vec3 a_Tangent;
12 layout (location = 4) in vec3 a_Bitangent;
13
14 uniform mat4 u_Model;
15 uniform mat4 u_View;
16 uniform mat4 u_Projection;
17
18 out vec3 v_FragPos;
19 out vec3 v_Normal;
20 out vec2 v_TexCoord;
21 out mat3 v_TBN;
22
23 void main() {
24 v_FragPos = vec3(u_Model * vec4(a_Position, 1.0));
25 v_Normal = mat3(transpose(inverse(u_Model))) * a_Normal;
26 v_TexCoord = a_TexCoord;
27
28 // Calculate TBN matrix for normal mapping
29 vec3 T = normalize(mat3(u_Model) * a_Tangent);
30 vec3 B = normalize(mat3(u_Model) * a_Bitangent);
31 vec3 N = normalize(mat3(u_Model) * a_Normal);
32 v_TBN = mat3(T, B, N);
33
34 gl_Position = u_Projection * u_View * vec4(v_FragPos, 1.0);
35 }
36 )";
37
38 const char* defaultFragmentShader = R"(
39 #version 330 core
40
41 struct Material {
42 vec4 albedo;
43 float metallic;
44 float roughness;
45 float ambientOcclusion;
46 };
47
48 in vec3 v_FragPos;
49 in vec3 v_Normal;
50 in vec2 v_TexCoord;
51 in mat3 v_TBN;
52
53 uniform Material u_Material;
54
55 uniform sampler2D u_AlbedoMap;
56 uniform sampler2D u_NormalMap;
57 uniform sampler2D u_MetallicMap;
58 uniform sampler2D u_RoughnessMap;
59 uniform sampler2D u_AmbientOcclusionMap;
60
61 uniform bool u_HasAlbedoMap;
62 uniform bool u_HasNormalMap;
63 uniform bool u_HasMetallicMap;
64 uniform bool u_HasRoughnessMap;
65 uniform bool u_HasAOMap;
66
67 uniform vec3 u_ViewPos;
68
69 out vec4 FragColor;
70
71 void main() {
72 // Sample textures
73 vec4 albedo = u_HasAlbedoMap ? texture(u_AlbedoMap, v_TexCoord) : u_Material.albedo;
74 float metallic = u_HasMetallicMap ? texture(u_MetallicMap, v_TexCoord).r : u_Material.metallic;
75 float roughness = u_HasRoughnessMap ? texture(u_RoughnessMap, v_TexCoord).r : u_Material.roughness;
76 float ambientOcclusion = u_HasAOMap ? texture(u_AmbientOcclusionMap, v_TexCoord).r : u_Material.ambientOcclusion;
77
78 // Normal mapping
79 vec3 normal = v_Normal;
80 if (u_HasNormalMap) {
81 normal = texture(u_NormalMap, v_TexCoord).rgb;
82 normal = normal * 2.0 - 1.0;
83 normal = normalize(v_TBN * normal);
84 }
85
86 // Simple lighting (можно заменить на PBR в будущем)
87 vec3 lightDir = normalize(vec3(1.0, 1.0, 1.0));
88 vec3 viewDir = normalize(u_ViewPos - v_FragPos);
89 vec3 reflectDir = reflect(-lightDir, normal);
90
91 float diff = max(dot(normal, lightDir), 0.0);
92 float spec = pow(max(dot(viewDir, reflectDir), 0.0), 32.0);
93
94 vec3 ambient = vec3(0.1) * albedo.rgb;
95 vec3 diffuse = diff * albedo.rgb;
96 vec3 specular = spec * vec3(0.5);
97
98 vec3 result = (ambient + diffuse + specular) * ambientOcclusion;
99 FragColor = vec4(result, albedo.a);
100 }
101 )";
102
103 m_Shader = Shader::Create(defaultVertexShader, defaultFragmentShader);
104}
