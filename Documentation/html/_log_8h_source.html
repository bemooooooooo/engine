127/**
128 * @brief Enables logging to a file.
129 *
130 * Redirects log output to the specified file. Can either append to
131 * an existing file or overwrite it.
132 *
133 * @param path The file path to log to
134 * @param append True to append to existing file, false to overwrite
135 * @return bool True if file logging was successfully enabled, false otherwise
136 * @example
137 * @code
138 * if (Log::EnableFileLogging("engine.log", true)) {
139 * ENGINE_INFO << "File logging enabled.";
140 * } else {
141 * ENGINE_ERROR << "Failed to enable file logging.";
142 * }
143 * @endcode
144 * @note If enabling file logging fails, the log output remains directed to console.
145 * @note Call DisableFileLogging() to stop logging to the file and restore console output.
146 */
147inline bool EnableFileLogging(const std::string& path = "engine.log", bool append = true) {
148 try {
149 std::ios_base::openmode mode = std::ios::out;
150 if (append) mode |= std::ios::app;
151 s_LogFile = CreateScope<std::ofstream>(path, mode);
152 if (!s_LogFile->is_open()) {
153 s_LogFile.reset();
154 return false;
155 }
156 s_OldCoutBuf = std::cout.rdbuf();
157 s_OldCerrBuf = std::cerr.rdbuf();
158 s_OldClogBuf = std::clog.rdbuf();
159
160 std::cout.rdbuf(s_LogFile->rdbuf());
161 std::cerr.rdbuf(s_LogFile->rdbuf());
162 std::clog.rdbuf(s_LogFile->rdbuf());
163 return true;
164 } catch (...) {
165 s_LogFile.reset();
166 return false;
167 }
168}

170/**
171 * @brief Disables file logging and restores console output.
172 *
173 * Stops logging to the file and restores the original console output streams.
174 * Closes the log file if it was open.
175 *
176 * @example
177 * @code
178 * Log::DisableFileLogging();
179 * ENGINE_INFO << "File logging disabled, back to console.";
180 * @endcode
181 */
182inline void DisableFileLogging() {
183 if (s_LogFile) {
184 if (s_OldCoutBuf) std::cout.rdbuf(s_OldCoutBuf);
185 if (s_OldCerrBuf) std::cerr.rdbuf(s_OldCerrBuf);
186 if (s_OldClogBuf) std::clog.rdbuf(s_OldClogBuf);
187 s_LogFile->close();
188 s_LogFile.reset();
189 }
190}
